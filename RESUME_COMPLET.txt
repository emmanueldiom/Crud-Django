🎯 DESCRIPTION GÉNÉRALE:
────────────────────────
Application Django complète permettant la gestion totale d'une base de données 
d'employés. Vous pouvez:
  • Ajouter des employés

  • Supprimer avec confirmation

  ✅ Serveur:  Django Dev Server (port 8000)

models.py - Modèle Employe:
────────────────────────────

from django.db import models

class Employe(models.Model):
    nom = models.CharField(max_length=100)
    email = models.EmailField()
    poste = models.CharField(max_length=100)
    salaire = models.DecimalField(max_digits=10, decimal_places=2)
    
    def __str__(self):
        return self.nom


views.py - Les 4 Opérations CRUD:
──────────────────────────────────

# READ - Liste
def liste_employes(request):
    employes = Employe.objects.all()
    return render(request, 'employe/list.html', {'employes': employes})

# CREATE - Ajouter
def ajouter_employe(request):
    form = EmployeForm(request.POST or None)
    if form.is_valid():
        form.save()
        return redirect('liste_employes')
    return render(request, 'employe/formulaire.html', {'form': form})

# UPDATE - Modifier
def modifier_employe(request, id):
    employe = get_object_or_404(Employe, id=id)
    form = EmployeForm(request.POST or None, instance=employe)
    if form.is_valid():
        form.save()
        return redirect('liste_employes')
    return render(request, 'employe/formulaire.html', {'form': form})

# DELETE - Supprimer
def supprimer_employe(request, id):
    employe = get_object_or_404(Employe, id=id)
    if request.method == "POST":
        employe.delete()
        return redirect('liste_employes')
    return render(request, 'employe/confirmer_suppression.html', 
                  {'employe': employe})
